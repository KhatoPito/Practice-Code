	class SampleReverseWordInSentence
	{
		static string ReverseSentence(string sentence)
		{
			char[] strArr = sentence.ToCharArray();
			Swap(0, strArr.Length -1 , ref strArr);


			for (int i = 0, j = 0; i <= strArr.Length - 1; i++)
			{
				int temp = i;

				if (strArr[i].Equals(' ') || (i + 1) == strArr.Length)
				{
					if (i < strArr.Length - 1)
					{
						Swap(j, i-1, ref strArr);
					}
					else
					{
						Swap(j, i, ref strArr);
					}

					j = temp + 1;
	
				}
			}

			return new string(strArr);
		}


		static void Swap(int l, int r, ref char[] arr)
		{
			while (l < r)
			{
				arr[l] ^= arr[r];
				arr[r] ^= arr[l];
				arr[l] ^= arr[r];

				l++;
				r--;
			}
		}

		static void Main(string[] args)
		{
			string sentence ="Hello World - I am Vedant";
			Console.WriteLine("The Original Sentence is: " + sentence);
			Console.WriteLine("The Reversed Sentence is: " + ReverseSentence(sentence));

			Console.ReadLine();

		}
	}
